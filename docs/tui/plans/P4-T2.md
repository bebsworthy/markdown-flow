## §2 Module design

### 2.1 Pure types — `src/browser/types.ts`

```ts
import type {
  ValidationDiagnostic,
  WorkflowDefinition,
} from "markflow";
import type { RegistryEntry } from "../registry/types.js";

/** Where the entry actually lives, once resolved. */
export type EntrySourceKind = "file" | "workspace";

/**
 * Status of a resolved registry entry.
 *   - "pending"      — async resolution in flight (initial render).
 *   - "valid"        — parsed cleanly + no `error`-severity diagnostics.
 *   - "parse-error"  — file exists but parse or validate failed hard.
 *   - "missing"      — file/dir not found (renders "✗ 404").
 */
export type EntryStatus = "pending" | "valid" | "parse-error" | "missing";

/**
 * Minimal last-run projection sourced from `RunManager.listRuns()`.
 * Only shapes we actually render in the list row.
 */
export interface LastRunInfo {
  /** "complete" | "error" | "running" | "suspended" — mirrors engine `RunStatus`. */
  readonly status: "complete" | "error" | "running" | "suspended";
  /** ISO-8601 UTC. Used to compute the "2h / 1d" badge. */
  readonly endedAt: string | null;
}

/**
 * An entry resolved for display. Carries enough info for both the list
 * row (status + badge + last-run) and the preview pane (parsed workflow
 * + diagnostics). Never mutated — resolver returns a fresh instance.
 */
export interface ResolvedEntry {
  /** The original registry record (pointer-equal to registryState.entries[i]). */
  readonly entry: RegistryEntry;
  /** Stable client-side id used for `SELECT_WORKFLOW` dispatch. See §2.3. */
  readonly id: string;
  /** Where the entry lives; `"file"` even for missing .md entries. */
  readonly sourceKind: EntrySourceKind;
  /** Resolved absolute path on disk (if resolvable); null for missing. */
  readonly absolutePath: string | null;
  /** Resolution status. See `EntryStatus` above. */
  readonly status: EntryStatus;
  /** Human-readable title — `workflow.name` when parsed, fallback to basename. */
  readonly title: string;
  /** Full parsed workflow when `status === "valid"`. null otherwise. */
  readonly workflow: WorkflowDefinition | null;
  /**
   * Diagnostics from `validateWorkflow` plus parser diagnostics. Empty
   * array when `status === "valid"` with zero diagnostics. Non-empty +
   * `status === "parse-error"` when parsing failed (synthetic diagnostic).
   */
  readonly diagnostics: ReadonlyArray<ValidationDiagnostic>;
  /** Last run info if any; null otherwise. */
  readonly lastRun: LastRunInfo | null;
  /** Opaque reason string when `status === "missing"` or `"parse-error"`. */
  readonly errorReason: string | null;
}

/** Options passed to `resolveEntry` / `resolveEntries`. */
export interface ResolverOptions {
  /** Base dir used to resolve relative `entry.source` values (typically cwd). */
  readonly baseDir: string;
  /** When true, query `RunManager.listRuns()` for last-run info. Defaults to true. */
  readonly readLastRun?: boolean;
  /** Override clock for deterministic "N hours ago" tests. Defaults to `Date.now`. */
  readonly now?: () => number;
}
```

### 2.2 Preview formatters — `src/browser/preview-layout.ts`

```ts
import type {
  FlowEdge,
  InputDeclaration,
  ValidationDiagnostic,
  WorkflowDefinition,
} from "markflow";
import type { ResolvedEntry } from "./types.js";

/**
 * Returns the "## Inputs" block lines. Leading "## Inputs" header is
 * returned as line[0]; body is indented two spaces. Empty array when the
 * workflow declares no inputs (so caller can omit the whole section).
 *
 * Format (matches mockups.md §2 line 61):
 *    ## Inputs
 *      sha      required   commit to deploy
 *      regions  default:us,eu
 *
 * The `name` column is right-padded to the longest-name width; the second
 * column is "required" | "default:VAL"; the third is `description`.
 */
export function formatInputsSummary(
  inputs: ReadonlyArray<InputDeclaration>,
): ReadonlyArray<string>;

/**
 * Returns the "## Flow" block as a list of lines. Leading "## Flow"
 * header is line[0]; body is indented two spaces. Walks the workflow
 * graph from nodes with `isStart` (or nodes with no incoming edges)
 * and emits one line per chain. forEach / fan-out annotations render
 * on indented continuation lines:
 *    build → test → review(approval) → publish
 *                    └─▶ forEach regions: deploy
 *
 * Caps at 6 body lines; excess collapses to "  … (N more edges)".
 * Returns [] for an empty graph.
 */
export function formatFlowSummary(
  workflow: WorkflowDefinition,
): ReadonlyArray<string>;

/**
 * Returns a one-line-per-diagnostic formatter. Format:
 *    ✗ FOREACH_NO_COLLECTOR  deploy-us: batch has no collector
 *    ⚠ UNREACHABLE           cleanup: no incoming edges
 * `✗` for severity "error", `⚠` for "warning". Glyph selection is driven
 * by the caller passing in a glyph table (not imported here — purity).
 * Signature accepts a plain string for each glyph to keep this file pure:
 */
export function formatDiagnostics(
  diagnostics: ReadonlyArray<ValidationDiagnostic>,
  glyphs: { readonly error: string; readonly warning: string },
): ReadonlyArray<string>;

/**
 * Counts steps, approvals, and forEach edges for the preview footer:
 *    "9 steps · 1 approval · 1 forEach"
 * forEach count = edges with `annotations.forEach` set.
 * approvals = steps where `type === "approval"`.
 */
export function countSteps(workflow: WorkflowDefinition): {
  readonly steps: number;
  readonly approvals: number;
  readonly forEach: number;
};

/**
 * Formats the one-line step-count sentence. Omits ` · N approvals` and
 * ` · N forEach` when counts are zero. "steps" always rendered.
 *   countSteps({steps: 9, approvals: 1, forEach: 1})
 *     → "9 steps · 1 approval · 1 forEach"
 *   countSteps({steps: 3, approvals: 0, forEach: 0})
 *     → "3 steps"
 */
export function formatStepCountLine(counts: ReturnType<typeof countSteps>): string;

/**
 * Source badge for the left-column row: "[file]" | "[workspace]".
 * Pure — just looks at `resolved.sourceKind`. Status ("parse"/"404")
 * is separately handled by `formatStatusFlag` below.
 */
export function formatSourceBadge(resolved: ResolvedEntry): string;

/**
 * Formats the validity/last-run flag column. Examples:
 *   valid + last-run "complete" 2h ago  → "✓ 2h"
 *   valid + last-run "error" 3d ago     → "✗ 3d"
 *   valid + no runs                     → "— never"
 *   parse-error                         → "✗ parse"
 *   missing                             → "✗ 404"
 * Returns { text, tone } where `tone` is "good" | "bad" | "neutral" —
 * the caller maps `tone` to theme colors (good → success, bad → danger,
 * neutral → dim).
 */
export function formatStatusFlag(
  resolved: ResolvedEntry,
  now?: number,
): { readonly text: string; readonly tone: "good" | "bad" | "neutral" };

/**
 * Short duration formatter: milliseconds → "45s" / "2h" / "3d". For
 * anything ≥ 7d emits "Nd". Used by `formatStatusFlag`.
 */
export function formatDurationShort(ms: number): string;

/** Returns `resolved.entry.source` canonicalised to use as a SELECT_WORKFLOW id. */
export function formatEntryId(entry: { readonly source: string }): string;
```

All formatters are **deterministic** — tests pass a fixed `now` where timestamps matter (`formatStatusFlag` second arg).

### 2.4 Resolver — `src/browser/resolver.ts`

```ts
import { readFile, stat } from "node:fs/promises";
import { basename, extname, resolve as resolvePath } from "node:path";
import {
  parseWorkflow,
  validateWorkflow,
  createRunManager,
  type WorkflowDefinition,
  type ValidationDiagnostic,
} from "markflow";
import { formatEntryId } from "./preview-layout.js";
import type {
  EntrySourceKind,
  LastRunInfo,
  ResolvedEntry,
  ResolverOptions,
} from "./types.js";
import type { RegistryEntry } from "../registry/types.js";

/**
 * Resolve a single registry entry.
 *
 * Algorithm:
 *   1. Compute `absolutePath = path.resolve(opts.baseDir, entry.source)`.
 *   2. `stat(absolutePath)`:
 *        ENOENT       → status = "missing", errorReason = "404".
 *        other errors → status = "missing", errorReason = err.message.
 *        dir          → sourceKind = "workspace". Go to (3a).
 *        .md file     → sourceKind = "file".      Go to (3b).
 *        other file   → status = "parse-error", errorReason = "not a .md".
 *   3a. workspace: try `readFile(path.join(absolutePath, ".markflow.json"))`.
 *        - If present, look for `workflowPath` key; parse THAT .md file.
 *          (Format inferred from features.md §3.1 URL-materialisation para;
 *          falls back to any *.md in the dir if `.markflow.json` has none.)
 *        - If missing → status = "parse-error", errorReason = "no workflow".
 *   3b. file: `await parseWorkflow(absolutePath)` in try/catch; on throw
 *        set status = "parse-error" with synthetic diagnostic.
 *   4. If parsed cleanly, call `validateWorkflow(workflow)`. Collect
 *      `parserDiagnostics` + validator diagnostics. status = "valid"
 *      iff zero `severity === "error"` diagnostics; otherwise
 *      "parse-error" (the preview still renders; only the list flag
 *      shows "✗ parse").
 *   5. If `opts.readLastRun !== false` and `sourceKind === "workspace"`,
 *      call `createRunManager(absolutePath).listRuns()` and pick the
 *      most recent run (max by `startedAt`) to populate `lastRun`. For
 *      `sourceKind === "file"`, the workflow has no dedicated run dir
 *      — `lastRun` stays null.
 *   6. Title derivation:
 *        - parsed workflow → `workflow.name` (engine's name field).
 *        - parse failure   → `path.basename(absolutePath)` (or source
 *                            verbatim for "missing").
 *   7. Compute `id = formatEntryId(entry)` — stable per-entry key.
 *   8. Return the `ResolvedEntry`.
 *
 * Never throws. Errors always become a structured `ResolvedEntry`.
 */
export async function resolveEntry(
  entry: RegistryEntry,
  opts: ResolverOptions,
): Promise<ResolvedEntry>;

/**
 * Batch convenience: resolves entries concurrently with `Promise.all`.
 * Caller is responsible for any cancellation (this task does not wire
 * AbortSignal; deferred to a later perf task).
 */
export async function resolveEntries(
  entries: ReadonlyArray<RegistryEntry>,
  opts: ResolverOptions,
): Promise<ReadonlyArray<ResolvedEntry>>;

/**
 * Construct a synthetic parse-error diagnostic from a thrown error so
 * the resolver's return shape stays uniform.
 */
export function synthesizeParseDiagnostic(
  err: unknown,
  sourceFile: string,
): ValidationDiagnostic;
```

### 2.5 Client-side id derivation

Since `RegistryEntry` carries no stable `id` field, the browser module derives one client-side:

```ts
formatEntryId({ source: "./flows/deploy.md" }) === "./flows/deploy.md"
```

This matches the existing `AppState.selectedWorkflowId` type (`string | null`) and avoids a reducer-action extension. Collisions are impossible because `addEntry` enforces `isSameSource` dedupe. Dispatched with `dispatch({ type: "SELECT_WORKFLOW", workflowId: id })`.

## §3 Component design

### 3.1 `WorkflowBrowser` (`src/components/workflow-browser.tsx`)

```tsx
export interface WorkflowBrowserProps {
  readonly registryState: RegistryState;
  readonly registryConfig: RegistryConfig;
  /** Current selection id from AppState.selectedWorkflowId. */
  readonly selectedWorkflowId: string | null;
  /** Dispatches SELECT_WORKFLOW + (TODO P4-T3) add/remove actions. */
  readonly dispatch: (action: Action) => void;
  /** Pane width override for tests. Defaults to 140. */
  readonly width?: number;
  /** Pane height override for tests. Defaults to app-shell top-slot rows. */
  readonly height?: number;
  /** Override resolver for tests (injects deterministic resolved entries). */
  readonly resolver?: (
    entries: ReadonlyArray<RegistryEntry>,
  ) => Promise<ReadonlyArray<ResolvedEntry>>;
}
```

Responsibilities:

1. Compute a sorted entry list via `sortByAddedAt(registryState.entries)` (most-recent first — matches the mockup ordering).
2. On mount and when `registryState.entries` identity changes: call `resolveEntries(entries, opts)` and stash the result in `useState<ReadonlyArray<ResolvedEntry>>`. Use a `useEffect` with a `cancelled` flag so late returns don't setState on unmounted components.
3. Render branch:
   - `entries.length === 0` → `<WorkflowBrowserEmpty width={width}/>` filling the whole `top` slot.

## §9 Risks / pitfalls

1. **Resolver fs I/O discipline.** `resolver.ts` is the single file in this task that touches disk. All fs access goes through `node:fs/promises`. The `resolveEntry` function NEVER throws — errors always become a structured `ResolvedEntry` with `status: "missing"` or `"parse-error"`. Tests assert this contract (no `expect.rejects`).
2. **Per-test tmpdir discipline (BLOCKING).** `test/browser/resolver.test.ts` must use `mkdtempSync(join(tmpdir(), "markflow-browser-"))` + `rmSync(..., {recursive, force})`. Never write outside tmpdir. Pattern inherited from P4-T1 `store.test.ts`.
3. **Engine `.markflow.json` format ambiguity.** features.md §3.1 references "`.markflow.json`" inside workspaces but doesn't spell out the key schema. **Default pick:** read it as plain JSON; look for a `workflowPath` string key (relative to the workspace dir). If absent, fall back to the first `*.md` file in the directory via `readdir`. Document as §11 Q1. If the engine later formalises a schema, a follow-up task updates the resolver.
4. **`RunManager.listRuns()` over a directory with no `runs/` subdir.** Engine source at `packages/markflow/src/core/run-manager.ts:205` returns `[]` gracefully. Tests confirm the no-runs path.
5. **`useEffect` + late-resolve setState after unmount.** Classic React hazard. Guard with `let cancelled = false; ...; return () => { cancelled = true; };`. Every setState inside the async handler must check `if (cancelled) return;`.
6. **Mermaid-as-text summary cyclic graphs.** `formatFlowSummary` walks edges; a cyclic workflow could recurse forever. **Mitigation:** maintain a `visited: Set<string>` during the walk; on re-visit, emit `(cycle)` marker and continue. If the total line count would exceed 6, collapse to `… (N edges, too large to preview)`. Tested in §4.1.
7. **Width prop defaulting.** If neither `width` prop nor `useStdout().stdout.columns` resolves, default to 140 inside `WorkflowBrowser`. Matches P3-T4 precedent (keybar defaults to 80 for min content).
8. **Theme literals.** Reviewer must grep for `color="red"` / `color="green"` / raw hex in any new `.tsx` file; all color props must be `color={theme.colors.X}`. No exceptions. The frame splitter (`│`) glyph comes from `theme.frame.v`, not a literal.
9. **`ink-testing-library` terminal width.** Width override prop required (see P3-T4 §3.1). Tests always pass explicit width.
10. **F-key / arrow-key handling in tests.** Arrow keys via raw ANSI `"\x1b[A"` / `"\x1b[B"` through `stdin.write` after `setImmediate`. `Enter` sends `"\r"`.
11. **Reducer purity.** Any new action type this task introduces WOULD break reducer purity tests. **Mitigation: this task introduces NO reducer actions.** It reuses `SELECT_WORKFLOW`. Verified: no edits to `src/state/**` or `test/state/reducer.test.ts`.
12. **Strict TS, no `any`.** `ResolvedEntry` and `LastRunInfo` are strict types. The engine's `WorkflowDefinition` is imported as `type` only; the resolver carries it unchanged. Diagnostics use the engine's `ValidationDiagnostic` type verbatim.
13. **`app.tsx` async load race.** If the user immediately presses `↓` before `loadRegistry` resolves, the browser renders with `entries: []` (empty-state). No crash. On late resolution, the component re-renders with entries. Tested indirectly via the `registryLoaded` flag (preview shows a brief "Loading..." under the list title bar for ≤ one frame).
14. **ESM + `.js` suffix imports.** Every relative import uses explicit `.js` suffixes. `import { ... } from "./preview-layout.js";`.
15. **Engine type imports via `markflow`.** Pure modules `preview-layout.ts` / `types.ts` import types from `"markflow"`. This is a **type-only import** (`import type { ... }` or inline `type` qualifier) — NOT a runtime import. Verified by the purity probe's `FORBIDDEN` regex set (type-only imports pass through the scan since no `fs/react/ink` regex matches `markflow`).
16. **Graph walker entry points.** `WorkflowDefinition.graph` carries `nodes: Map` + `edges: FlowEdge[]`. Start nodes = nodes with `isStart === true` OR nodes with no incoming edges. `formatFlowSummary` prefers `isStart` when any exists; else uses the incoming-edge heuristic.
17. **`.markflow.json` reader pattern.** Uses the same pattern as the registry store: `try { readFile; JSON.parse; } catch { return { kind: "no-workspace-config" }; }`. Never throws.
